{
    int a[10] ,i,j,k,swap;
    for(i=0;i<=9;i++)

        cin>>a[i];
    }
      for(i=0;i<=9;i++)
      {
           for(j=i+1;j<=9;j++)

              if(a[i]<a[j])
                {

                }
